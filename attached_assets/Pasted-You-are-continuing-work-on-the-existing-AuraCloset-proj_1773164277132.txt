
### 5. Call to Google Cloud Vision (Label Detection)

- Use `axios.post` to call this endpoint.
- After receiving the response, read `responses[0].labelAnnotations`. Each label has fields including `description` and `score`.

### 7. Pseudo-code / expected implementation of /classify-garment

Implement something equivalent to this logic (you can adjust names/style but behaviour must match):

```js
app.post("/classify-garment", async (req, res) => {
  try {
    const { imageBase64, imageUrl, userId } = req.body;

    if ((!imageBase64 && !imageUrl) || (imageBase64 && imageUrl)) {
      return res.status(400).json({ error: "imageBase64 or imageUrl required" });
    }

    const visionReq = {
      requests: [
        {
          image: imageBase64
            ? { content: imageBase64 }
            : { source: { imageUri: imageUrl } },
          features: [{ type: "LABEL_DETECTION", maxResults: 10 }]
        }
      ]
    };

    const apiKey = process.env.GCV_API_KEY;
    if (!apiKey) {
      return res.status(500).json({ error: "missing_gcv_api_key" });
    }

    const visionRes = await axios.post(
      `https://vision.googleapis.com/v1/images:annotate?key=${apiKey}`,
      visionReq
    );

    const annotations =
      visionRes.data.responses?.[0]?.labelAnnotations || [];

    const labels = annotations.map(l => ({
      description: l.description,
      score: l.score
    }));

    const CONF_THRESHOLD = 0.7;
    let garmentType = "unknown";
    let modelConfidence = 0;

    for (const l of labels) {
      if (l.score > modelConfidence) {
        modelConfidence = l.score;
      }
      const mapped = GARMENT_LABEL_MAP[l.description];
      if (mapped && l.score >= CONF_THRESHOLD) {
        garmentType = mapped;
        modelConfidence = l.score;
        break;
      }
    }

    let colorFamily = null;
    for (const l of labels) {
      const mappedColor = COLOR_LABEL_MAP[l.description];
      if (mappedColor) {
        colorFamily = mappedColor;
        break;
      }
    }

    return res.json({
      garmentType,
      colorFamily,
      modelConfidence,
      rawLabels: labels.slice(0, 5),
      source: "gcv_label_detection"
    });
  } catch (err) {
    console.error("Vision error", err.response?.data || err.message);
    return res.status(500).json({ error: "classification_failed" });
  }
});
```
